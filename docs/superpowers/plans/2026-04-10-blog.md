# Blog Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a blog to telemetrydrops.com with Astro content collections, optimized for traditional SEO and LLM citation.

**Architecture:** Astro content collections for markdown blog posts in `src/content/blog/`. Static pages generated at build time via `src/pages/blog/` routes. Two-column post layout with sticky TOC sidebar. Card grid index with client-side tag filtering. JSON-LD structured data for SEO and AI-SEO on every post.

**Tech Stack:** Astro content collections, Shiki (built-in), `@astrojs/rss`, `remark-gfm`, `rehype-autolink-headings`, reading time remark plugin, Tailwind CSS.

**Spec:** `docs/superpowers/specs/2026-04-10-blog-design.md`

```javascript
import getReadingTime from 'reading-time';
import { toString } from 'mdast-util-to-string';

export function remarkReadingTime() {
  return function (tree, { data }) {
    const textOnPage = toString(tree);
    const readingTime = getReadingTime(textOnPage);
    data.astro.frontmatter.minutesRead = Math.ceil(readingTime.minutes);
  };
}
```

- [ ] **Step 3: Create content collection schema**

Create `src/content.config.ts`:

```typescript
import { defineCollection, z } from "astro:content";
import { glob } from "astro/loaders";
import { VALID_TAGS } from "./data/tag-data";
import { authors } from "./data/author-data";

const authorKeys = Object.keys(authors) as [string, ...string[]];

const blog = defineCollection({
  loader: glob({ pattern: "**/*.md", base: "./src/content/blog" }),
  schema: z.object({
    title: z.string(),
    slug: z.string(),
    description: z.string(),
    publishedAt: z.coerce.date(),
    updatedAt: z.coerce.date().optional(),
    author: z.enum(authorKeys),
    tags: z.array(z.enum(VALID_TAGS)).min(1),
    series: z.string().optional(),
    seriesOrder: z.number().optional(),
    ogImage: z.string().optional(),
    readingTime: z.number().optional(),
    tldr: z.string(),
    keyTakeaways: z.array(z.string()).min(1),
    faqEntries: z
      .array(
        z.object({
          question: z.string(),
          answer: z.string(),
        })
      )
      .optional(),
  }),
});

export const collections = { blog };
```

```astro
  title: string;
  description: string;
  slug: string;
  publishedAt: Date;
  updatedAt?: Date;

const techArticle = {
  "@context": "https://schema.org",
  "@type": "TechArticle",
  headline: title,
  description: description,
  datePublished: publishedAt.toISOString(),
  dateModified: (updatedAt ?? publishedAt).toISOString(),
  author: {
    "@type": "Person",
    name: author.name,
    jobTitle: author.role,
    url: author.social.linkedin ?? author.social.github ?? "",
    sameAs: [author.social.linkedin, author.social.github, author.social.bluesky].filter(Boolean),
  },
  publisher: {
    "@type": "Organization",
    name: "TelemetryDrops",
    url: "https://telemetrydrops.com",
    logo: "https://telemetrydrops.com/logo.svg",
  },
  keywords: tags,
  url: canonicalUrl,
  mainEntityOfPage: canonicalUrl,
};

const faqPage = faqEntries && faqEntries.length > 0
  ? {
      "@context": "https://schema.org",
      "@type": "FAQPage",
      mainEntity: faqEntries.map((entry) => ({
        "@type": "Question",
        name: entry.question,
        acceptedAnswer: {
          "@type": "Answer",
          text: entry.answer,
        },
      })),
    }
  : null;
---

<script type="application/ld+json" set:html={JSON.stringify(techArticle)} />
```

```astro
    </div>

          <a
            href={`#${heading.slug}`}
            class:list={[
              "block border-l-2 -ml-px py-1 transition-colors hover:text-telemetria-orange hover:border-telemetria-orange",
              heading.depth === 2 ? "pl-4 text-white/60 border-transparent" : "pl-8 text-white/40 border-transparent",
            ]}
            data-toc-link
          >
            {heading.text}
          </a>
        </li>
      ))}
    </ul>
  </nav>
)}

<script>
  function setupTocHighlighting() {
    const tocLinks = document.querySelectorAll('[data-toc-link]');
    if (tocLinks.length === 0) return;

    const headingIds = Array.from(tocLinks).map((link) => {
      const href = link.getAttribute('href');
      return href ? href.slice(1) : '';
    });

    const observer = new IntersectionObserver(
      (entries) => {
        for (const entry of entries) {
          if (entry.isIntersecting) {
            const id = entry.target.id;
            tocLinks.forEach((link) => {
              const href = link.getAttribute('href');
              if (href === `#${id}`) {
                link.classList.add('text-telemetria-orange', 'border-telemetria-orange');
                link.classList.remove('text-white/60', 'text-white/40', 'border-transparent');
              } else {
                link.classList.remove('text-telemetria-orange', 'border-telemetria-orange');
                link.classList.add('border-transparent');
                // Restore original opacity based on depth
                if (link.classList.contains('pl-4')) {
                  link.classList.add('text-white/60');
                } else {
                  link.classList.add('text-white/40');
                }
              }
            });
          }
        }
      },
      { rootMargin: '-80px 0px -60% 0px', threshold: 0 }
    );

    headingIds.forEach((id) => {
      const el = document.getElementById(id);
      if (el) observer.observe(el);
    });
  }

  document.addEventListener('DOMContentLoaded', setupTocHighlighting);
  document.addEventListener('astro:page-load', setupTocHighlighting);
</script>
```
